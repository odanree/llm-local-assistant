## 🎯 Key Implementation Points

### The "Prompt Detector" Circuit Breaker

This is the breakthrough that prevents the "Retry Trap":

```typescript
// In executeRun() method
handle.onData(chunk => {
  stdout += chunk;

  // ← THE CIRCUIT BREAKER ←
  if (detectPrompt(chunk)) {
    // Immediately suspend - do NOT auto-retry
    return {
      requiresUserInput: true,
      promptText: extractPromptText(chunk),
      suggestedInputs: getSuggestedInputs(chunk)
    };
  }

  // Emit streaming output
  this.config.onStreamData?.(step.stepId, chunk, false);
});
```

**Key insight**: We detect the prompt while data arrives, set requiresUserInput flag, and return immediately. The executor's main loop checks this flag and transitions to SUSPENDED_FOR_PERMISSION (not FAILED, not retrying).

### Structure

```typescript
import * as cp from 'child_process';
import { ProcessHandle, SpawnOptions } from '../types';
import { ICommandRunner } from './ICommandRunner';
import { detectPrompt, looksLikePrompt } from '../types/StreamingIO';

/**
 * AsyncCommandRunner: Implements ProcessHandle via child_process.spawn()
 *
 * Replaces exec() for interactive commands (npm, yarn, git, etc.)
 * Provides real-time streaming and interactive stdin support.
 */
export class AsyncCommandRunner implements ICommandRunner {
  private processes: Map<number, ProcessHandle> = new Map();
  private nextPID: number = 1000;

  /**
   * Execute a command and return all output at once
   * (Existing behavior - unchanged for backward compatibility)
   */
  async exec(command: string): Promise<{ stdout: string; stderr: string }> {
    // Keep existing impl - delegate to spawn() if needed
  }

  /**
   * NEW v2.12.0: Spawn a process and return ProcessHandle
   *
   * Usage:
   * ```typescript
   * const handle = runner.spawn('npm install', { timeout: 30000 });
   * handle.onData(data => console.log(data));
   * handle.onExit(code => console.log('done:', code));
   * handle.sendInput('y\n');
   * handle.kill();
   * ```
   */
  spawn(command: string, options: SpawnOptions = {}): ProcessHandle {
    const cwd = options.cwd || process.cwd();
    const shell = options.shell ?? true;
    const timeout = options.timeout ?? 30000;
    const maxBuffer = options.maxBuffer ?? 10 * 1024 * 1024;

    // Parse command for spawn()
    const [cmd, ...args] = this.parseCommand(command, shell);

    // Spawn the child process
    const child = cp.spawn(cmd, args, {
      cwd,
      shell,
      stdio: ['pipe', 'pipe', 'pipe'], // stdin, stdout, stderr
      timeout // Will auto-kill after this many ms
    });

    // Create ProcessHandle implementation
    const handle: ProcessHandle = {
      onData: () => {}, // Will be set by caller
      onError: () => {},
      onExit: () => {},
      onPrompt: () => {},
      sendInput: () => {},
      kill: () => {},
      isRunning: () => true,
      getExitCode: () => undefined,
      getPID: () => child.pid,
      stdin: child.stdin,
      stdout: child.stdout,
      stderr: child.stderr
    };

    // Setup event handlers and timeout logic
    this.setupHandlers(child, handle, timeout, options.echo);

    return handle;
  }

  /**
   * Internal: Parse command string into [cmd, ...args]
   *
   * Examples:
   * - "npm install" → ["npm", "install"]
   * - "npm install react" → ["npm", "install", "react"]
   * - "cd src && npm install" → ["sh", "-c", "cd src && npm install"]
   */
  private parseCommand(
    command: string,
    shell: boolean | string
  ): [string, ...string[]] {
    if (typeof shell === 'string') {
      // Custom shell specified
      return [shell, '-c', command];
    }

    if (!shell) {
      // No shell - split by spaces
      const parts = command.trim().split(/\s+/);
      return parts as [string, ...string[]];
    }

    // Default shell (bash on Unix, cmd.exe on Windows)
    const isWindows = process.platform === 'win32';
    const shellCmd = isWindows ? 'cmd.exe' : '/bin/sh';
    const arg = isWindows ? '/c' : '-c';
    return [shellCmd, arg, command];
  }

  /**
   * Internal: Setup all event handlers
   *
   * This is the core streaming logic:
   * 1. Collect stdout/stderr into buffers
   * 2. Emit real-time data events
   * 3. Detect prompts and emit onPrompt
   * 4. Manage timeout and cleanup
   * 5. Track process state
   */
  private setupHandlers(
    child: cp.ChildProcess,
    handle: ProcessHandle,
    timeout: number,
    echo: boolean = false
  ): void {
    let stdout = '';
    let stderr = '';
    let exited = false;
    let timeoutId: NodeJS.Timeout | null = null;
    let promptDetected = false;

    const dataCallbacks: ((data: string) => void)[] = [];
    const errorCallbacks: ((error: Error) => void)[] = [];
    const exitCallbacks: ((code: number) => void)[] = [];
    const promptCallbacks: ((text: string) => void)[] = [];

    // Register callbacks
    handle.onData = (cb) => dataCallbacks.push(cb);
    handle.onError = (cb) => errorCallbacks.push(cb);
    handle.onExit = (cb) => exitCallbacks.push(cb);
    handle.onPrompt = (cb) => promptCallbacks.push(cb);

    // Handle stdout data
    child.stdout?.on('data', (chunk) => {
      const data = chunk.toString();
      stdout += data;

      // Check if this looks like a prompt
      if (!promptDetected && looksLikePrompt(data)) {
        promptDetected = true;
        const pattern = detectPrompt(data);
        if (pattern) {
          promptCallbacks.forEach(cb => cb(data));
        }
      }

      // Emit to caller
      dataCallbacks.forEach(cb => cb(data));
    });

    // Handle stderr data
    child.stderr?.on('data', (chunk) => {
      const data = chunk.toString();
      stderr += data;

      // Emit to caller (marked as error)
      errorCallbacks.forEach(cb => cb(new Error(data)));
    });

    // Handle process exit
    child.on('exit', (code) => {
      exited = true;
      if (timeoutId) clearTimeout(timeoutId);

      // Emit final exit code
      exitCallbacks.forEach(cb => cb(code ?? 1));

      // Cleanup
      handle.isRunning = () => false;
      handle.getExitCode = () => code ?? undefined;
    });

    // Handle process errors
    child.on('error', (err) => {
      exited = true;
      if (timeoutId) clearTimeout(timeoutId);

      errorCallbacks.forEach(cb => cb(err));
      exitCallbacks.forEach(cb => cb(1));
    });

    // Implement sendInput
    handle.sendInput = (data: string) => {
      if (child.stdin && !child.stdin.destroyed) {
        child.stdin.write(data);
        if (echo) {
          dataCallbacks.forEach(cb => cb(`[USER INPUT] ${data}`));
        }
      }
    };

    // Implement kill with SIGTERM → SIGKILL escalation
    handle.kill = () => {
      if (exited) return;

      // Send SIGTERM first
      child.kill('SIGTERM');

      // If still alive after 5s, send SIGKILL
      timeoutId = setTimeout(() => {
        if (!exited && child.kill) {
          child.kill('SIGKILL');
        }
      }, 5000);
    };

    // Implement isRunning
    handle.isRunning = () => !exited;

    // Setup auto-timeout
    if (timeout > 0) {
      timeoutId = setTimeout(() => {
        if (!exited) {
          handle.kill();
          errorCallbacks.forEach(cb =>
            cb(new Error(`Process timeout after ${timeout}ms`))
          );
        }
      }, timeout);
    }
  }
}

### 3. Update Executor for Streaming

**File**: `src/executor.ts` (executeRun method)

```typescript
private async executeRun(step: ExecutionStep): Promise<StepResult> {
  // Use spawn() instead of exec() for interactive commands
  const handle = this.commandRunner.spawn(step.command!, {
    cwd: this.config.workspace.fsPath,
    timeout: this.config.timeout || 30000
  });

  let stdout = '';
  let stderr = '';
  let requiresInput = false;
  let promptText = '';

  // Stream output in real-time
  handle.onData(data => {
    stdout += data;
    this.config.onStreamData?.(step.stepId, data, false);

    // ← THE CIRCUIT BREAKER ←
    // Detect prompt and immediately flag as requiring input
    if (!requiresInput && detectPrompt(data)) {
      requiresInput = true;
      promptText = extractPromptText(data);
      // Don't kill the process - just return
    }
  });

  handle.onError(err => {
    stderr += err.message;
    this.config.onStreamData?.(step.stepId, err.message, true);
  });

  // Return immediately if prompt detected (don't wait for exit)
  if (requiresInput) {
    return {
      stepId: step.stepId,
      success: false,
      output: stdout,
      requiresUserInput: true,
      promptText,
      suggestedInputs: getSuggestedInputs(promptText),
      duration: Date.now() - stepStartTime
    };
  }

  // Otherwise wait for process to exit
  return new Promise((resolve) => {
    handle.onExit(code => {
      resolve({
        stepId: step.stepId,
        success: code === 0,
        output: stdout,
        error: code !== 0 ? stderr : undefined,
        duration: Date.now() - stepStartTime
      });
    });
  });
}
